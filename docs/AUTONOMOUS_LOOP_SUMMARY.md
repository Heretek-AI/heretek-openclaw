- Agent discovery with 5-minute cache
- Local registry fallback when A2A unavailable

**Key Features:**
```javascript
class A2AClient {
    // Primary: Try LiteLLM A2A endpoints
    // Fallback: Use Redis pub/sub for message delivery
    // Timeout: 500ms for A2A, then fall back to Redis
    async sendMessage(toAgent, message, options = {}) {
        const a2aResult = await this._tryA2A(toAgent, messagePayload);
        if (a2aResult.success) return a2aResult.response;
        
        // Fall back to Redis
        console.log(`[A2AClient] A2A failed for ${toAgent}, falling back to Redis`);
        return await this._tryRedis(toAgent, messagePayload);
    }
}
```

class LiberationShield {
    constructor(config = {}) {
        this.config = {
            mode: config.mode || 'transparent', // transparent = audit without blocking
            enablePromptInjectionDetection: true,
            enableJailbreakDetection: true,
            enableAnomalyDetection: true,
            enableAuditLogging: true,
            maxLogEntries: 10000
        };
    }
    
    async protect(operation, context) {
        // Analyze input
        const inputAnalysis = await this.analyzeInput(operation.input, context);
        
        // Check for anomalies
        const anomalyCheck = await this.checkAnomaly(operation, context);
        
        // Build result
        const result = {
            allowed: inputAnalysis.safe && !anomalyCheck.anomalous,
            sanitizedInput: inputAnalysis.sanitized,
            mode: this.config.mode
        };
        
        // Execute if allowed (or in transparent mode)
        if (result.allowed || this.config.mode === 'transparent') {
            result.result = operation.execute ? await operation.execute(...) : null;
        }
        
        return result;
    }
}
